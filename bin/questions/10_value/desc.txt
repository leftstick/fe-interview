## 提示：
### 你的`value`模块应该完成如下功能：

```javascript
var value = require('value');

var what = 'The fuck!';

console.log(value(what)); //The fuck!

var hello = function() {
    return 'world';
};

console.log(value(hello)); //world

var tripleFn = function() {
    return function(){
        return function(){
            return 987654;
        }
    };
};

console.log(value(tripleFn)); //987654
```
